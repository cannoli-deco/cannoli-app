void loadPost(List<DocumentSnapshot> postData) {
  posts.clear();
  for (int i = 0; i < postData.length; i++) {
    // TODO: Date needs to be change
    posts.add(feedPost(postData[i].data()["owner_name"], postData[i].data()["content"], DateTime.now()));
  }
}
